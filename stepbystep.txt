const fs = require('fs');

// Crear un ReadableStream desde un archivo
const readableStream = fs.createReadStream('archivo.txt');

// Crear un Blob a partir del stream
const blob = new Blob([readableStream], { type: 'application/octet-stream' });

// Crear una URL para el Blob
const url = URL.createObjectURL(blob);

// Crear un enlace para descargar
const link = document.createElement('a');
link.href = url;
link.download = 'archivo_descargado.txt';

// Agregar el enlace al documento
document.body.appendChild(link);

// Hacer clic en el enlace para iniciar la descarga
link.click();
